-- TEMA 1
-- EX 1
SELECT A.NUME, COUNT(DISTINCT C.ID_ARTIST_DESCHIDERE)
FROM ARTISTI A
JOIN TURNEE T ON A.ID_ARTIST = T.ID_ARTIST
JOIN CONCERTE C ON C.ID_TURNEU = T.ID_TURNEU
GROUP BY A.NUME;

-- EX 2
SELECT A.NUME, COUNT(ID_ALBUM)
FROM ARTISTI A
JOIN (
    SELECT T.ID_ARTIST
    FROM CONCERTE C
    JOIN TURNEE T ON C.ID_TURNEU = T.ID_TURNEU
    WHERE EXTRACT(YEAR FROM C.DATA) = 2024
    GROUP BY T.ID_ARTIST
    HAVING COUNT(C.ID_CONCERT) >= 3
    ) MINIM_CONCERTE ON A.ID_ARTIST = MINIM_CONCERTE.ID_ARTIST
JOIN (
    SELECT ID_ARTIST, MAX(AN_DECERNARE) AS ULT_PREM
    FROM PREMII_CASTIGATE P2
    GROUP BY P2.ID_ARTIST
    ) P ON A.ID_ARTIST = P.ID_ARTIST
JOIN ALBUME AL ON AL.ID_ARTIST = A.ID_ARTIST AND AL.AN_LANSARE < P.ULT_PREM
GROUP BY A.NUME;

-- EX 3
SELECT T.ID_ARTIST, A.NUME, SUM(C.VENIT)
FROM CONCERTE C
JOIN TURNEE T ON T.ID_TURNEU = C.ID_TURNEU
JOIN ARTISTI A ON A.ID_ARTIST = T.ID_ARTIST
WHERE EXTRACT(YEAR FROM C.DATA) = 2025 AND
      C.ID_ARTIST_DESCHIDERE IN
            (
            -- ARTISTII CARE AU CASTIGAT MINIM 3 PREMII:
            SELECT A2.ID_ARTIST
            FROM ARTISTI A2
            JOIN PREMII_CASTIGATE P ON A2.ID_ARTIST = P.ID_ARTIST
            GROUP BY A2.ID_ARTIST, A2.NUME
            HAVING COUNT(DISTINCT P.ID_PREMIU) >= 3
            )
GROUP BY T.ID_ARTIST, A.NUME;
